post_install do |installer|
  installer.pods_project.targets.each do |target|

      phase.shell_script = 'echo "xcframework skipped"; exit 0'
    end
  end
end
`;
      fs.writeFileSync(podfilePath, podfile + "\n" + hook);
      return config;
    },
  ]);
